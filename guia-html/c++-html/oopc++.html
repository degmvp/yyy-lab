Guia de Referência de Orientação a Objetos em C++

1. Descrição da Classe em C++

Em C++, uma classe é um tipo de dado definido pelo usuário que contém:

  • Atributos/Membros dados - variáveis que armazenam o estado do objeto
  • Métodos/Funções membro - funções que definem o comportamento do objeto
  • Construtores e Destrutores - para inicialização e limpeza
  • Modificadores de acesso - public, private e protected

Estrutura básica de uma classe:

class NomeDaClasse {
private:
    // Atributos privados
    tipo atributo1;
    tipo atributo2;
    
public:
    // Construtor
    NomeDaClasse(tipo param1, tipo param2) {
        // inicialização
    }
    
    // Métodos públicos
    tipo metodo1(tipo parametro) {
        // implementação
    }
    
    tipo metodo2() {
        // implementação
    }
    
    // Destrutor
    ~NomeDaClasse() {
        // limpeza
    }
};

2. Exemplo Prático: Cadastro de Produtos e Fornecedores

2.1 Classe Produto

#include <iostream>
#include <string>

using namespace std;

class Produto {
private:
    int id;
    string nome;
    string descricao;
    double preco;
    int quantidadeEstoque;
    int idFornecedor;

public:
    // Construtor
    Produto(int id, string nome, string descricao, double preco, int quantidadeEstoque, int idFornecedor) {
        this->id = id;
        this->nome = nome;
        this->descricao = descricao;
        this->preco = preco;
        this->quantidadeEstoque = quantidadeEstoque;
        this->idFornecedor = idFornecedor;
    }

    // Getters
    int getId() { return id; }
    string getNome() { return nome; }
    string getDescricao() { return descricao; }
    double getPreco() { return preco; }
    int getQuantidadeEstoque() { return quantidadeEstoque; }
    int getIdFornecedor() { return idFornecedor; }

    // Setters
    void setNome(string nome) { this->nome = nome; }
    void setDescricao(string descricao) { this->descricao = descricao; }
    void setPreco(double preco) { this->preco = preco; }
    void setQuantidadeEstoque(int quantidade) { this->quantidadeEstoque = quantidade; }
    void setIdFornecedor(int id) { this->idFornecedor = id; }

    // Método para exibir informações do produto
    void exibirInformacoes() {
        cout << "ID: " << id << endl;
        cout << "Nome: " << nome << endl;
        cout << "Descrição: " << descricao << endl;
        cout << "Preço: R$ " << preco << endl;
        cout << "Quantidade em estoque: " << quantidadeEstoque << endl;
        cout << "ID Fornecedor: " << idFornecedor << endl;
    }

    // Método para atualizar estoque
    void atualizarEstoque(int quantidade) {
        quantidadeEstoque += quantidade;
        if (quantidadeEstoque < 0) quantidadeEstoque = 0;
    }
};

2.2 Classe Fornecedor

class Fornecedor {
private:
    int id;
    string nome;
    string endereco;
    string telefone;
    string email;

public:
    // Construtor
    Fornecedor(int id, string nome, string endereco, string telefone, string email) {
        this->id = id;
        this->nome = nome;
        this->endereco = endereco;
        this->telefone = telefone;
        this->email = email;
    }

    // Getters
    int getId() { return id; }
    string getNome() { return nome; }
    string getEndereco() { return endereco; }
    string getTelefone() { return telefone; }
    string getEmail() { return email; }

    // Setters
    void setNome(string nome) { this->nome = nome; }
    void setEndereco(string endereco) { this->endereco = endereco; }
    void setTelefone(string telefone) { this->telefone = telefone; }
    void setEmail(string email) { this->email = email; }

    // Método para exibir informações do fornecedor
    void exibirInformacoes() {
        cout << "ID: " << id << endl;
        cout << "Nome: " << nome << endl;
        cout << "Endereço: " << endereco << endl;
        cout << "Telefone: " << telefone << endl;
        cout << "Email: " << email << endl;
    }
};

2.3 Exemplo de uso das classes

int main() {
    // Criando fornecedores
    Fornecedor fornecedor1(1, "Distribuidora ABC", "Rua das Flores, 123", "(11) 1234-5678", "contato@abc.com");
    Fornecedor fornecedor2(2, "Atacadão XYZ", "Av. Principal, 456", "(11) 9876-5432", "vendas@xyz.com");

    // Criando produtos
    Produto produto1(101, "Notebook", "Notebook i5 8GB RAM 512GB SSD", 3500.00, 10, 1);
    Produto produto2(102, "Mouse sem fio", "Mouse óptico sem fio 1600DPI", 120.50, 25, 2);

    // Exibindo informações
    cout << "=== Fornecedores ===" << endl;
    fornecedor1.exibirInformacoes();
    cout << endl;
    fornecedor2.exibirInformacoes();
    cout << endl;

    cout << "=== Produtos ===" << endl;
    produto1.exibirInformacoes();
    cout << endl;
    produto2.exibirInformacoes();
    cout << endl;

    // Atualizando estoque
    produto1.atualizarEstoque(-2); // Venda de 2 unidades
    produto2.atualizarEstoque(10); // Compra de 10 unidades

    cout << "=== Estoque atualizado ===" << endl;
    cout << "Estoque do Notebook: " << produto1.getQuantidadeEstoque() << endl;
    cout << "Estoque do Mouse: " << produto2.getQuantidadeEstoque() << endl;

    return 0;
}

3. Definição de Métodos e Funções

3.1 Métodos

Métodos são funções definidas dentro de uma classe que operam nos dados da classe. Eles definem o comportamento dos objetos.

Tipos comuns de métodos:

  • Construtores - Inicializam o objeto
  • Destrutores - Liberam recursos
  • Getters - Retornam valores de atributos
  • Setters - Modificam valores de atributos
  • Métodos de negócio - Implementam a lógica da aplicação

3.2 Exemplos Práticos

Método Construtor

Produto::Produto(int id, string nome, string descricao, double preco, int quantidadeEstoque, int idFornecedor) {
    this->id = id;
    this->nome = nome;
    this->descricao = descricao;
    this->preco = preco;
    this->quantidadeEstoque = quantidadeEstoque;
    this->idFornecedor = idFornecedor;
}

Métodos Getter e Setter

// Getter
string Produto::getNome() {
    return nome;
}

// Setter
void Produto::setNome(string nome) {
    this->nome = nome;
}

Método de Negócio

void Produto::atualizarEstoque(int quantidade) {
    quantidadeEstoque += quantidade;
    if (quantidadeEstoque < 0) {
        quantidadeEstoque = 0;
    }
}

Método para Exibição

void Fornecedor::exibirInformacoes() {
    cout << "ID: " << id << endl;
    cout << "Nome: " << nome << endl;
    cout << "Endereço: " << endereco << endl;
    cout << "Telefone: " << telefone << endl;
    cout << "Email: " << email << endl;
}

4. Tabelas SQL para Produtos e Fornecedores

4.1 MySQL

Tabela Fornecedores

CREATE TABLE fornecedores (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    endereco VARCHAR(200),
    telefone VARCHAR(20),
    email VARCHAR(100)
);

Tabela Produtos

CREATE TABLE produtos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    descricao TEXT,
    preco DECIMAL(10, 2) NOT NULL,
    quantidade_estoque INT DEFAULT 0,
    id_fornecedor INT,
    FOREIGN KEY (id_fornecedor) REFERENCES fornecedores(id)
);

4.2 SQL Server

Tabela Fornecedores

CREATE TABLE fornecedores (
    id INT IDENTITY(1,1) PRIMARY KEY,
    nome NVARCHAR(100) NOT NULL,
    endereco NVARCHAR(200),
    telefone NVARCHAR(20),
    email NVARCHAR(100)
);

Tabela Produtos

CREATE TABLE produtos (
    id INT IDENTITY(1,1) PRIMARY KEY,
    nome NVARCHAR(100) NOT NULL,
    descricao NVARCHAR(MAX),
    preco DECIMAL(10, 2) NOT NULL,
    quantidade_estoque INT DEFAULT 0,
    id_fornecedor INT,
    FOREIGN KEY (id_fornecedor) REFERENCES fornecedores(id)
);

6. Relacionamento entre Classes e Tabelas

O relacionamento entre as classes Produto e Fornecedor é representado:

  • No código C++: através do atributo idFornecedor na classe Produto
  • No banco de dados: através da chave estrangeira id_fornecedor na tabela produtos

Este é um exemplo de relacionamento 1-N (um para muitos), onde um fornecedor pode fornecer muitos produtos, mas cada produto tem apenas um fornecedor.
